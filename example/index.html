notivuecation

Promise-based alert/confirm modal for Vue.js

basic alert/confirm

this.$confirm('Are you sure you want to do this?');
this.$alert('Please click ok');

custom alert/confirm

this.$confirm({
  message: 'Are you sure you want to do this?',
  confirm: 'Yes!',
  cancel: 'Hell no',
});

this.$alert({
  title: 'Important!',
  message: 'Please confirm',
  confirm: 'Sure!',
});

fully custom

this.$notify({
  title: 'Custom title',
  message: 'Custom message',
  buttons: [
    {label: 'option #1', value: 1},
    {label: 'option #2', value: 2},
    {label: 'option #3', value: 3},
  ]
})
